1function [lp]=gaussmixp(y,m,v,w)
 2% y = cat(1, testSamples(1).mfcc{:});
 3% m = gmm.M;

 7%GAUSSMIXP calculate probability densities from a Gaussian mixture model
 8%
 9% Inputs: n data values, k mixtures, p parameters, q data vector size
10%
11%   Y(n,q) = input data
12%   M(k,p) = mixture means for x(p)
13%   V(k,p) or V(p,p,k) variances (diagonal or full)
14%   W(k,1) = weights
15%   A(q,p), B(q) = transformation: y=x*a'+b' (where y and x are row vectors)
16%            if A is omitted, it is assumed to be the first q rows of the
17%            identity matrix. B defaults to zero.
18%   Note that most commonly, q=p and A and B are omitted entirely.
19%
20% Outputs
21%
22%  LP(n,1) = log probability of each data point
23%  RP(n,k) = relative probability of each mixture
24%  KH(n,1) = highest probability mixture
25%  KP(n,1) = relative probability of highest probability mixture

49[n,q]=size(y);
50[k,p]=size(m);
51memsize=voicebox('memsize');    % set memory size to use
52
53lp=zeros(n,1);
54wk=ones(k,1);
55
56vi=-0.5*v.^(-1);                % data-independent scale factor in exponent
57lvm=log(w)-0.5*sum(log(v),2);   % log of external scale factor (excluding -0.5*q*log(2pi) term)
58ii=1:n;
59wnj=ones(1,n);
60kk=repmat(ii,k,1);
61km=repmat(1:k,1,n);
62py=reshape(sum((y(kk(:),:)-m(km(:),:)).^2.*vi(km(:),:),2),k,n)+lvm(:,wnj);
63mx=max(py,[],1);                % find normalizing factor for each data point to prevent underflow when using exp()
64px=exp(py-mx(wk,:));            % find normalized probability of each mixture for each datapoint
65ps=sum(px,1);                   % total normalized likelihood of each data point
66lp(ii)=log(ps)+mx;
67lp=lp-0.5*q*log(2*pi);
